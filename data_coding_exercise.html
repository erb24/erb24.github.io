#Plots

#Since np.linalg.eigh sorts the eigenvalues by DESCENDING variance,
#I want the last eigvector for the first PC.
PC1 = xi[:,-1]
#One-dimensional free-energy surfaces along the first two PCs
his1, bins = np.histogram(PC1, bins = 10)
dx = bins[1] - bins[0]

#Save bin-width as a separate variable for later use in the mutual information calculation
dx1 = dx

#Get the mid-point of each bin
bins = (bins[1:] + bins[:-1]) / 2

#Normalize the probability density
his1 = his1 / (dx*his1.sum())

#Get the free-energy by taking the log of the probability density
fes1 = -np.log(his1)

#Gauge transformation to set the minimum of the free-eenrgy to 0
fes1 = fes1 - fes1.min()

#Plot it
plt.plot(bins, fes1, lw = 2, c = 'k')
plt.fill_between(bins, 0, fes1, color = 'grey', alpha = 0.5)
plt.xlabel('PC 1', fontsize = 16)
plt.ylabel(r'Free energy, k$_B$T', fontsize = 16)
plt.xticks(size = 12)
plt.yticks(size = 12)
plt.xlim((bins[0], bins[-1]))
plt.ylim((0, 1.1*fes1.max()))
plt.tight_layout()
plt.savefig('PC1_fes.pdf', dpi = 300)
plt.show()
plt.close()

#Mutual information
#Since the PCs are continuous variables, I need to use the probability density functions 
#and the bin-widths.

MI = 0
for i in range(his12.shape[0]):
    for j in range(his12.shape[1]):
        #Skip un-sampled points in the two-dimensional distribution
        if his12[i,j] == 0:
            pass
        else:
            MI += his12[i, j] * np.log2(his12[i, j] / (his1[i]*his2[j])) * dx1 * dx2
print('Mutual information (bits): ', MI)
